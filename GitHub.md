# Create a new repository on the command line:

>	git init 

>	git add < files >

>	git commit -m "first commit"

>	git branch -M main

>	git remote add origin https://github.com/AlexanderFur/obsidian_notes.git

>	git push -u origin main

# Everyday use:
>	git status 

>	git pull
